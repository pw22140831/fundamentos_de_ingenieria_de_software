  -id: int
  -nombreCompleto: string
  -correo: string
  -password: string
  -rol: string
  +iniciarSesion()
}

class Administrador {
  +registrarUsuario()
  +asignarRol()
  +revocarRol()
  +generarReportes()
  +gestionarProveedores()
}

class Operador {
  +registrarTransaccion()
  +gestionarInventario()
  +consultarInventario()
}

class Trabajador {
  +registrarTransaccion()
}

  -idProyecto: int

  -fechaFin: date
  -presupuesto: double
  +calcularRentabilidad()
}

class Transaccion {
  -idTransaccion: int
  -tipo: string
  -monto: double
  -fecha: date
  -descripcion: string
  +validar()
}


class Proveedor {
  -idProveedor: int
  -nombre: string
  -contacto: string
  +editar()
}

class Inventario {
  -idInventario: int
  -nombreRecurso: string
  -cantidad: int
  -estado: string
  +actualizarCantidad()
}

class Auditoria {
  -idAuditoria: int
  -accion: string
  -fechaHora: datetime
  -modulo: string
}
